}); } } catch(e) { Logger.log('CasaDados erro: '+e); } // API 2: BrasilAPI try { let url = 'https://brasilapi.com.br/api/cnpj/v1/search?search='+encodeURIComponent(nome||cidade); if(uf) url += '&uf='+uf; if(cidade) url += '&municipio='+encodeURIComponent(cidade); const r = UrlFetchApp.fetch(url, {muteHttpExceptions:true}); if(r.getResponseCode()===200){ const arr = JSON.parse(r.getContentText()); (Array.isArray(arr)?arr:(arr.data||[])).forEach(function(e){ if(!lista.find(function(x){return (x.cnpj||'').replace(/\D/g,'')===(e.cnpj||'').replace(/\D/g,'');})) lista.push(e); }); } } catch(e) { Logger.log('BrasilAPI erro: '+e); } // API 3: ReceitaWS try { const q = nome||(cidade?cidade+' '+uf:''); let url = 'https://receitaws.com.br/v1/cnpj/search?query='+encodeURIComponent(q); (Array.isArray(d)?d:(d.data||[])).forEach(function(e){ if(!lista.find(function(x){return (x.cnpj||'').replace(/\D/g,'')===(e.cnpj||'').replace(/\D/g,'');})) lista.push(e); }); } } catch(e) { Logger.log('CNPJws erro: '+e); if(rows.length < 2) return {ok:false, msg:'Nenhum usuario cadastrado'}; const hdrs = rows[0].map(function(h){ return String(h).toLowerCase().trim(); }); for(var i=1; i 1) sheet.getRange(2,1,lastRow-1,sheet.getLastColumn()).clearContent(); const rows = sheet.getDataRange().getValues(); if(rows.length < 2) return {ok:true, data:[]}; const hdrs = rows[0].map(function(h){ return String(h).toLowerCase().trim(); }); return {ok:true, data: rows.slice(1).map(function(r){ return toObj(hdrs,r); })}; } function toObj(headers, row) { var obj = {}; headers.forEach(function(h,i){ obj[h]=row[i]; }); return obj; }
